define function add
with parameters a b
begin
    return a + b
end function

create integer i
create integer total
create float average

set i to 0
set total to call add with 2 3 * 4

while i less than 3 do
    set total to total + i
    set i to i + 1
end while

for i from 1 to 2 step 1 do
    set total to total + i
end for

if total greater than 10 then
    set average to cast total to float / 2.0
    print average
else
    print total
end if

if i equals 3 then
    print i
else
    print total
end if
